# Stage 1: Build
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy csproj và restore dependencies
COPY ["Shorten.Redirect/Shorten.Redirect.csproj", "Shorten.Redirect/"]
COPY ["Shorten.Data/Shorten.Data.csproj", "Shorten.Data/"]
RUN dotnet restore "Shorten.Redirect/Shorten.Redirect.csproj"

# Copy toàn bộ source code
COPY . .

# Build và publish
WORKDIR "/src/Shorten.Redirect"
RUN dotnet publish -c Release -o /app/publish

# Stage 2: Runtime (image nhỏ hơn, chỉ chứa runtime)
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS final
WORKDIR /app
COPY --from=build /app/publish .

EXPOSE 8080
ENTRYPOINT ["dotnet", "Shorten.Redirect.dll"]
